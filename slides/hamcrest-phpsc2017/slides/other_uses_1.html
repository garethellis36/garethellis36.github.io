function foo(array $arrayOfObjects)
{
    foreach ($arrayOfObjects as $object) {
        if (! $object instanceof MyObject) {
            throw new \InvalidArgumentException(
                "Object is not an instance of " . MyObject::class
            );
        }
    }
}
    

    
function foo(array $arrayOfObjects)
{
    assertThat($arrayOfObjects, everyItem(is(anInstanceOf(MyObject::class)));
    // throws \Hamcrest\AsssertionError if assertion fails
}
